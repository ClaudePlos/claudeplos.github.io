-- Jezeli nie otwieraja sie KARTY PRACY - to pewnie podwójne wpisy - sleect mozna rozbudowac np. dla danego obiekti a nie konkretnego grafiku
select count(ks_prc_id), ks_prc_id from NGR_KARTY_SZCZEGOLY WHERE KS_PRC_ID IN (
SELECT DISTINCT PRACOWNIK_ID id FROM NGR_WIERSZE_GRAFIKOW WHERE GRAFIK_ID IN (
SELECT ID FROM NGR_GRAFIKI
WHERE NAZWA LIKE '%Z119%TRANSPORT%'
AND OKRES_ID = 26
AND USUNIETY = 'N'
AND UZ = 'N'
)
)
and KS_OKRES_ID = 28
AND ZAKONCZONA = 'N'
group by ks_prc_id

-- sprawdzenie pzu, wyrownanie zaleglych skladek
select lst_numer, lst_data_obliczen, prc_nazwisko, prc_imie, sk_wartosc
from ek_skladniki, ek_listy, ek_def_skladnikow, ek_pracownicy
where lst_id=sk_lst_id
and sk_prc_id = prc_id
and dsk_id=sk_dsk_id
and dsk_kod = '434'
and lst_data_obliczen = '2010-09-01'
order by 3 desc

-- dublowane potracenia, komu potracili wiecej niz 1 raz
select count(prc_pesel), prc_pesel
from
(select nvl(prc_numer,'0') prc_numer, prc_pesel, prc_nazwisko, prc_imie, sk_prc_id
, to_char(lst_data_obliczen,'YYYY-MM') data_obliczen, lst_data_wyplaty
, sk_wartosc wart_pzu, lst_id, lst_drl_kod
from ek_listy, ek_skladniki, ek_pracownicy
where sk_lst_id=lst_id
and sk_prc_id=prc_id
and sk_dsk_id=102087
and sk_status = 1
and to_char(lst_data_obliczen,'YYYY-MM') = substr(:okres,1,4)||'-'||substr(:okres,5,2)) a
full outer join
(select * from nap_pzu where pzu_okres = substr(:okres,1,4)||'-'||substr(:okres,5,2)) b
on pzu_pesel=prc_pesel
and pzu_okres=data_obliczen
group by prc_pesel having count(prc_pesel) > 1
order by 1 desc
